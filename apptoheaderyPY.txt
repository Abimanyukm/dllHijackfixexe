input_file = "ASIRVAD-GOLDLOANMODULEUAT.exe"
output_header = "app_data.h"

with open(input_file, "rb") as f:
    data = f.read()

with open(output_header, "w") as out:
    out.write("unsigned char appData[] = {\n")
    for i, byte in enumerate(data):
        out.write(f"0x{byte:02x},")
        if (i + 1) % 16 == 0:
            out.write("\n")
    out.write("\n};\n")
    out.write(f"unsigned int appDataLen = {len(data)};\n")
